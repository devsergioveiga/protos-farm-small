```typescript
async function getNextAssetTag(tx: TxClient, organizationId: string): Promise<string> {
  const last = await tx.asset.findFirst({
    where: { organizationId },
    orderBy: { assetTag: 'desc' },
    select: { assetTag: true },
  });
  const lastNum = last?.assetTag ? parseInt(last.assetTag.replace('PAT-', ''), 10) : 0;
  return `PAT-${String(lastNum + 1).padStart(5, '0')}`;
}
```

```typescript
async function getNetBookValue(tx: TxClient, assetId: string): Promise<Decimal> {
  const asset = await tx.asset.findUniqueOrThrow({
    where: { id: assetId },
    select: { acquisitionValue: true },
  });
  const depAgg = await tx.depreciationEntry.aggregate({
    where: { assetId, reversedAt: null },
    _sum: { depreciationAmount: true },
  });
  return new Decimal(asset.acquisitionValue ?? 0).minus(
    new Decimal(depAgg._sum.depreciationAmount ?? 0),
  );
}
```

    **3. Service** -- Create `asset-leasings.service.ts`:

    - **`createLeasing(ctx, input)`**: Validates dates (endDate > startDate), required fields. Uses `prisma.$transaction(async (tx) => { ... })` (NOT withRlsContext). Inside transaction:
      1. `getNextAssetTag(tx, ctx.organizationId)` -- copy helper from asset-acquisitions.service.ts
      2. `tx.asset.create({ data: { organizationId, farmId, assetType: input.assetType, classification: 'DEPRECIABLE_CPC27', name: 'ROU — ${input.lessorName} — ${input.contractNumber ?? assetTag}', assetTag, acquisitionDate: new Date(input.startDate), acquisitionValue: String(input.totalContractValue), status: 'ATIVO', notes: 'Arrendamento mercantil CPC 06...' } })`
      3. Compute `durationMonths = monthsBetween(startDate, endDate)` using simple date diff: `(endYear - startYear) * 12 + (endMonth - startMonth)`
      4. `tx.depreciationConfig.create({ data: { organizationId, assetId: rouAsset.id, method: 'STRAIGHT_LINE', usefulLifeMonths: durationMonths, residualValue: '0' } })`
      5. Compute `monthlyInstallment = new Decimal(input.totalContractValue).dividedBy(input.installmentCount).toDecimalPlaces(2)`
      6. Generate installments: `generateInstallments(Money(input.totalContractValue), input.installmentCount, new Date(input.firstDueDate))`
      7. `tx.payable.create({ data: { organizationId, farmId, supplierName: input.lessorName, category: 'FINANCING', description: 'Leasing CPC 06 — ${rouAsset.assetTag}', totalAmount: Money(input.totalContractValue).toDecimal(), dueDate: installments[0].dueDate, installmentCount: input.installmentCount, originType: 'ASSET_LEASING', originId: rouAsset.id } })`
      8. `tx.payableInstallment.createMany({ data: installments.map(i => ({ payableId, number: i.number, amount: i.amount.toDecimal(), dueDate: i.dueDate })) })`
      9. `tx.assetLeasing.create({ ... })`
      10. Return `{ leasing, rouAsset, payableId }`
