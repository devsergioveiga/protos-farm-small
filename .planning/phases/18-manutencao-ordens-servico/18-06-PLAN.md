```typescript
async function submitRequest() {
  const payload = {
    assetId: form.assetId,
    title: form.title,
    description: form.description,
    type: 'SOLICITACAO',
    photoUrls: photoBase64 ? [photoBase64] : [],
    geoLat: location?.coords.latitude ?? null,
    geoLon: location?.coords.longitude ?? null,
  };

  // Save locally
  await saveLocalRequest({
    ...payload,
    id: uuid(),
    synced: false,
    createdAt: new Date().toISOString(),
  });

  // Enqueue for sync
  await enqueuePendingOperation({
    entity: 'maintenance_requests',
    operation: 'CREATE',
    payload,
    endpoint: '/api/org/work-orders',
    method: 'POST',
  });

  // Try immediate sync if online
  if (isConnected) {
    await flushNow();
  }

  // Show success toast and navigate back
  Alert.alert(
    'Solicitacao enviada',
    isConnected ? 'OS criada com sucesso.' : 'Sera enviada quando houver conexao.',
  );
  router.back();
}
```
