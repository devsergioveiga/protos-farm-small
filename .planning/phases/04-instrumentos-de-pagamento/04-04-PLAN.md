  totalAmount: number; status: string; payableId: string | null;
  expenses: ExpenseOutput[]; closedAt: string | null;
}

  category: string | null; notes: string | null;
}

API Endpoints:

- GET /org/credit-cards — list cards with currentBill
- GET /org/credit-cards/:id — card with bills
- POST /org/credit-cards — create card
- PUT /org/credit-cards/:id — update card
- DELETE /org/credit-cards/:id — soft delete
- POST /org/credit-cards/:id/expenses — add expense
- GET /org/credit-cards/:id/bills — list bills
- POST /org/credit-cards/bills/:billId/close — close bill, returns CP
