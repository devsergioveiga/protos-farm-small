1. Create `apps/frontend/src/hooks/useAssetBulkImport.ts`:

State machine: idle -> uploading -> mapping -> previewing -> confirming -> done

```typescript
export type ImportStep = 'idle' | 'uploading' | 'mapping' | 'previewing' | 'confirming' | 'done';

export interface ImportState {
  step: ImportStep;
  columnHeaders: string[];
  suggestedMapping: Record<string, string>;
  columnMapping: Record<string, string>;
  allRows: Record<string, string>[];
  validRows: any[];
  invalidRows: any[];
  result: { imported: number; skipped: number; failed: number; errors: any[] } | null;
  error: string | null;
  loading: boolean;
}

export function useAssetBulkImport() {
  // uploadFile(file: File): POST /import/parse with FormData, transition to mapping step
  // setMapping(mapping): update columnMapping
  // preview(): POST /import/preview with rows + columnMapping, transition to previewing step
  // confirm(): POST /import/confirm with validRows, transition to done step
  // downloadTemplate(): GET /import/template as blob download
  // reset(): back to idle
  return { state, uploadFile, setMapping, preview, confirm, downloadTemplate, reset };
}
```
