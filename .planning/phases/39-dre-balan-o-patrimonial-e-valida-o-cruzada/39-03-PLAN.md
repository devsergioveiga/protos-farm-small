```typescript
// GET /api/org/:orgId/financial-statements/balance-sheet?fiscalYearId=...&month=...
interface BpOutput {
  ativo: BpGroup[];   // [AC, ANC]
  passivo: BpGroup[]; // [PC, PNC, PL]
  totalAtivo: BpGroupRow;
  totalPassivo: BpGroupRow;
  indicators: BpIndicators;
}

interface BpGroup {
  id: string;         // 'ac' | 'anc' | 'pc' | 'pnc' | 'pl'
  label: string;
  rows: BpGroupRow[];
  total: BpGroupRow;
}

interface BpGroupRow {
  accountId: string | null;
  code: string;
  name: string;
  currentBalance: string;
  priorBalance: string;
  isSubtotal: boolean;
  level: number;
}

interface BpIndicators {
  liquidezCorrente: string | null;
  liquidezSeca: string | null;
  endividamentoGeral: string | null;
  composicaoEndividamento: string | null;
  roe: string | null;
  plPorHectare: string | null;
  sparklines: Record<string, { month: number; value: number }[]>;
}

// GET /api/org/:orgId/financial-statements/cross-validation?fiscalYearId=...&month=...
interface CrossValidationOutput {
  invariants: InvariantResult[];
  allPassed: boolean;
}

interface InvariantResult {
  id: string;
  title: string;
  status: 'PASSED' | 'FAILED' | 'PENDING';
  expected: string | null;
  found: string | null;
  difference: string | null;
  investigateUrl: string | null;
}
```
