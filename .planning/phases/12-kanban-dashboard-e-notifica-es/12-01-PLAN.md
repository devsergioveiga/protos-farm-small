```typescript
export const KANBAN_COLUMNS = [
  'RC_PENDENTE',
  'APROVADA',
  'EM_COTACAO',
  'OC_EMITIDA',
  'AGUARDANDO_ENTREGA',
  'RECEBIDO',
  'PAGO',
] as const;
export type KanbanColumn = (typeof KANBAN_COLUMNS)[number];

export interface KanbanCard {
  id: string; // purchaseRequestId (or purchaseOrderId for emergency POs)
  column: KanbanColumn;
  number: string; // RC sequential number or PO number for emergencies
  type: string; // RC requestType
  requester: string; // user display name
  totalValue: number; // sum of item estimated prices OR PO total
  urgency: 'NORMAL' | 'URGENTE' | 'EMERGENCIAL';
  daysInStage: number; // days since entering current column
  isOverdue: boolean; // SLA exceeded OR delivery date passed
  purchaseRequestId: string | null;
  quotationId: string | null;
  purchaseOrderId: string | null;
  goodsReceiptId: string | null;
  payableId: string | null;
  isEmergency: boolean;
}

export interface KanbanFilters {
  farmId?: string;
  urgency?: string;
  category?: string;
  supplierId?: string;
  startDate?: Date;
  endDate?: Date;
}

export const ALLOWED_TRANSITIONS: Record<KanbanColumn, KanbanColumn[]> = {
  RC_PENDENTE: ['APROVADA'],
  APROVADA: ['EM_COTACAO'],
  EM_COTACAO: ['OC_EMITIDA'],
  OC_EMITIDA: ['AGUARDANDO_ENTREGA'],
  AGUARDANDO_ENTREGA: ['RECEBIDO'],
  RECEBIDO: ['PAGO'],
  PAGO: [],
};

export class PurchaseKanbanError extends Error {
  constructor(
    message: string,
    public statusCode: number,
  ) {
    super(message);
    this.name = 'PurchaseKanbanError';
  }
}
```

Create `apps/backend/src/modules/purchase-kanban/purchase-kanban.service.ts`:

- `getKanbanCards(ctx: RlsContext, filters: KanbanFilters): Promise<KanbanCard[]>`:
  - Query 1: All PurchaseRequests with status NOT IN ('RASCUNHO', 'CANCELADA', 'REJEITADA') via `withRlsContext`. Include nested: `quotations` (where deletedAt null, orderBy createdAt desc, take 1, include purchaseOrders with nested goodsReceipts), `items` (select estimatedUnitPrice, quantity), `createdByUser` (select name). Apply optional filters (farmId, urgency, dateRange on createdAt).
  - Query 2: Emergency PurchaseOrders where `isEmergency=true` AND `quotationId=null` AND `deletedAt=null` AND status NOT IN ('CANCELADA'). Include `goodsReceipts`, `supplier`, `creator`, `items`.
  - Map each RC to a KanbanCard using `assignColumn(rc)` helper:
    - Check deepest pipeline entity present and its status:
      - If payableId from goodsReceipt exists and ALL PayableInstallments have status PAID -> PAGO
      - If GoodsReceipt with status CONFIRMADO exists -> RECEBIDO
      - If PurchaseOrder exists with status in (EMITIDA, CONFIRMADA, EM_TRANSITO) -> AGUARDANDO_ENTREGA (name reflects awaiting delivery, not just OC_EMITIDA)
      - If PurchaseOrder exists with status RASCUNHO or just created -> OC_EMITIDA
      - If Quotation exists -> EM_COTACAO
      - If RC status is APROVADA -> APROVADA
      - Otherwise (PENDENTE, EM_ANALISE) -> RC_PENDENTE
    - Compute `daysInStage`: difference in days between now and the timestamp of entering the current stage (use updatedAt of the deepest entity, or RC.submittedAt for pending)
    - Compute `isOverdue`: For RC stages check slaDeadline, for PO stages check expectedDeliveryDate
    - Compute `totalValue`: sum of items estimatedUnitPrice \* quantity (from RC), or PO total if PO exists
  - Map emergency POs similarly: start at OC_EMITIDA or deeper, set `isEmergency: true`, `purchaseRequestId: null`, urgency EMERGENCIAL

- `transitionCard(ctx: RlsContext & { userId: string }, cardId: string, targetColumn: KanbanColumn): Promise<void>`:
  - Load the card's current column by re-running assignColumn logic
  - Validate transition against ALLOWED_TRANSITIONS map
  - Throw PurchaseKanbanError(403) if transition not allowed: "Transicao nao permitida de {from} para {to}"
  - Dispatch the appropriate domain action based on targetColumn:
    - APROVADA: import and call approve from purchase-requests service
    - EM_COTACAO: throw PurchaseKanbanError(400, "Use a pagina de cotacoes para criar uma cotacao")
    - OC_EMITIDA: throw PurchaseKanbanError(400, "Use a pagina de cotacoes para aprovar e gerar pedido")
    - AGUARDANDO_ENTREGA: throw PurchaseKanbanError(400, "Use a pagina de pedidos para confirmar envio")
    - RECEBIDO: throw PurchaseKanbanError(400, "Use a pagina de recebimentos para confirmar")
    - PAGO: throw PurchaseKanbanError(400, "Use o contas a pagar para registrar pagamento")
  - NOTE: Only RC_PENDENTE -> APROVADA is a direct kanban action. Other transitions are informational — the user is directed to the correct page. This prevents the kanban from bypassing domain validation.
