    **Step 2: Create `useDfc.ts` hook** — Follow `useDre.ts` pattern exactly:
    ```typescript
    import { useState, useEffect, useCallback } from 'react';
    import { api } from '@/services/api';
    import type { DfcOutput } from '@/types/financial-statements';

    export function useDfc(orgId: string | undefined, fiscalYearId: string | null, month: number | null) {
      const [data, setData] = useState<DfcOutput | null>(null);
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState<string | null>(null);

      const fetchDfc = useCallback(async () => {
        if (!orgId || !fiscalYearId || !month) return;
        setLoading(true);
        setError(null);
        try {
          const response = await api.get(`/org/${orgId}/financial-statements/dfc`, {
            params: { fiscalYearId, month },
          });
          setData(response.data);
        } catch (err: any) {
          setError(err.response?.data?.error || 'Erro ao carregar DFC');
        } finally {
          setLoading(false);
        }
      }, [orgId, fiscalYearId, month]);

      useEffect(() => { fetchDfc(); }, [fetchDfc]);

      return { data, loading, error, refetch: fetchDfc };
    }
    ```
