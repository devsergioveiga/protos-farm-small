    ```typescript
      const [validationResult, setValidationResult] = useState<SpedValidationResult | null>(null);
      const [validationLoading, setValidationLoading] = useState(false);

      // --- Notes state ---
      const [notesText, setNotesText] = useState('');
      const [notesSaved, setNotesSaved] = useState(false);
      const [notesLoading, setNotesLoading] = useState(false);

      // --- Download state ---
      const [spedDownloading, setSpedDownloading] = useState(false);
      const [pdfDownloading, setPdfDownloading] = useState(false);

      // validate(fiscalYearId) — called on FY change
      // Calls GET /api/org/{orgId}/sped-ecd/validate?fiscalYearId={id}
      // Sets validationResult and validationLoading

      // downloadSped(fiscalYearId) — triggers file download
      // Calls GET /api/org/{orgId}/sped-ecd/download?fiscalYearId={id}
      // Creates blob URL, clicks <a>, revokes URL
      // On success: toast "Arquivo SPED_ECD_{CNPJ}_{ANO}.txt baixado com sucesso."
      // On error: toast "Nao foi possivel gerar o arquivo SPED. Tente novamente."

      // downloadPdf(fiscalYearId, costCenterId?) — triggers PDF download
      // Calls GET /api/org/{orgId}/integrated-report/download?fiscalYearId={id}&costCenterId={id}
      // Same blob download pattern
      // On success: toast "Relatorio integrado {ANO} gerado com sucesso."
      // On error: toast "Nao foi possivel gerar o relatorio PDF. Tente novamente."

      // loadNotes() — initial load
      // Calls GET /api/org/{orgId}/integrated-report/notes
      // Sets notesText from response

      // saveNotes(text) — debounced autosave
      // Calls PATCH /api/org/{orgId}/integrated-report/notes with { notesText: text }
      // On success: set notesSaved=true, after 3s set notesSaved=false
      // On error: toast "Nao foi possivel salvar as notas. Tente novamente."

      return {
        validationResult, validationLoading, validate,
        spedDownloading, downloadSped,
        pdfDownloading, downloadPdf,
        notesText, setNotesText, notesSaved, notesLoading, loadNotes, saveNotes,
      };
    }
    ```

    ```typescript
    const res = await fetch(url, { headers: { Authorization: `Bearer ${token}` } });
    ```

    ```tsx
    export default function SpedEcdPage() {
      const { orgId } = useAuth(); // or however orgId is obtained in this project
      const { fiscalYears } = useFiscalYears(orgId); // reuse from DrePage pattern
      const [selectedFyId, setSelectedFyId] = useState('');
      const [activeTab, setActiveTab] = useState<'sped' | 'report'>('sped');
      const [costCenterId, setCostCenterId] = useState('');

      const {
        validationResult, validationLoading, validate,
        spedDownloading, downloadSped,
        pdfDownloading, downloadPdf,
        notesText, setNotesText, notesSaved, notesLoading, loadNotes, saveNotes,
      } = useSpedEcd(orgId);

      // On fiscal year change: validate automatically (D-16)
      useEffect(() => {
        if (selectedFyId) { validate(selectedFyId); }
      }, [selectedFyId]);

      // Load notes on mount
      useEffect(() => { loadNotes(); }, []);

      return (
        <main id="main-content" className="sped-ecd-page">
          {/* Breadcrumb: Inicio / Contabilidade / SPED / Relatorios */}
          <nav aria-label="Caminho da pagina">
            <ol className="breadcrumb">
              <li><Link to="/">Inicio</Link></li>
              <li><Link to="/chart-of-accounts">Contabilidade</Link></li>
              <li>SPED / Relatorios</li>
            </ol>
          </nav>

          {/* Page header with FileText icon */}
          <header className="sped-ecd-page__header">
            <FileText size={24} />
            <h1>SPED ECD e Relatorios</h1>
          </header>

          {/* Fiscal year selector (shared between tabs, per D-03, D-21) */}
          <section className="sped-ecd-page__filter-bar" aria-label="Periodo">
            <div className="sped-ecd-page__filter-group">
              <label htmlFor="sped-fy-select">Exercicio Fiscal</label>
              <select
                id="sped-fy-select"
                value={selectedFyId}
                onChange={e => setSelectedFyId(e.target.value)}
              >
                <option value="">Selecione o exercicio</option>
                {fiscalYears.map(fy => (
                  <option key={fy.id} value={fy.id}>{fy.name}</option>
                ))}
              </select>
            </div>
          </section>

          {/* Tab rail — IMPORTANT: each button has id matching aria-labelledby on its panel */}
          <nav role="tablist" className="sped-ecd-page__tabs">
            <button
              id="tab-sped"
              role="tab"
              aria-selected={activeTab === 'sped'}
              aria-controls="panel-sped"
              onClick={() => setActiveTab('sped')}
              className={`sped-ecd-page__tab ${activeTab === 'sped' ? 'sped-ecd-page__tab--active' : ''}`}
            >
              SPED ECD
            </button>
            <button
              id="tab-report"
              role="tab"
              aria-selected={activeTab === 'report'}
              aria-controls="panel-report"
              onClick={() => setActiveTab('report')}
              className={`sped-ecd-page__tab ${activeTab === 'report' ? 'sped-ecd-page__tab--active' : ''}`}
            >
              Relatorio Integrado
            </button>
          </nav>

          {/* Tab panels using hidden attribute (per Phase 37/40 decision) */}
          <section id="panel-sped" role="tabpanel" aria-labelledby="tab-sped" hidden={activeTab !== 'sped'}>
            {!selectedFyId ? (
              <EmptyState /> // FileText icon 48px, "Selecione o exercicio fiscal", helper text
            ) : (
              <>
                <ValidationPanel result={validationResult} loading={validationLoading} />
                <div className="sped-ecd-page__action-bar">
                  <button
                    className="btn btn-primary"
                    disabled={!selectedFyId || validationResult?.hasErrors || spedDownloading}
                    aria-disabled={!selectedFyId || validationResult?.hasErrors}
                    title={validationResult?.hasErrors ? 'Corrija os erros acima para gerar o arquivo' : undefined}
                    onClick={() => downloadSped(selectedFyId)}
                  >
                    {spedDownloading ? 'Gerando arquivo...' : 'Gerar SPED ECD'}
                  </button>
                </div>
              </>

            {!selectedFyId ? (
              <EmptyState />
            ) : (
              <>
                {/* Farm selector (optional, per UI-SPEC) */}
                <div className="sped-ecd-page__filter-group">
                  <label htmlFor="farm-select">Fazenda (opcional)</label>
                  <select id="farm-select" value={costCenterId} onChange={e => setCostCenterId(e.target.value)}>
                    <option value="">Consolidado (todas as fazendas)</option>
                    {/* Map cost centers/farms here */}
                  </select>
                </div>

                <NotesTextarea
                  value={notesText}
                  onChange={setNotesText}
                  onSave={saveNotes}
                  saved={notesSaved}
                  loading={notesLoading}
                />

                <div className="sped-ecd-page__action-bar">
                  <button
                    className="btn btn-primary"
                    disabled={!selectedFyId || pdfDownloading}
                    onClick={() => downloadPdf(selectedFyId, costCenterId || undefined)}
                  >
                    {pdfDownloading ? 'Gerando PDF...' : 'Gerar Relatorio PDF'}
                  </button>
                </div>
              </>
            )}
          </section>
        </main>
      );
    }
    ```
