    export interface VacationCalcInput {
      baseSalary: Decimal;
      daysScheduled: number;

      bonusThird: Decimal;
      abonoValue: Decimal;
      grossTaxable: Decimal;
      inssAmount: Decimal;
      irrfAmount: Decimal;
      fgtsAmount: Decimal;
      netAmount: Decimal;
    }

    export interface ScheduleVacationInput {
      organizationId: string;
      employeeId: string;
      acquisitivePeriodId: string;
      startDate: string; // ISO date
      totalDays: number; // 5-30
      abono: number; // 0 or 10
      createdBy: string;
    }

    export interface VacationPeriodOutput { ... } // id, employeeId, startDate, endDate, daysEarned, daysTaken, balance, status, doublingDeadline, isNearDoubling

    **vacation-schedules.service.ts:**
    - `calculateVacationPay(input: VacationCalcInput, params: EngineParams): VacationCalcResult` — pure function following RESEARCH.md Pattern 2 exactly:
      - dailyRate = baseSalary / 30
      - vacationBase = dailyRate * daysScheduled
      - bonusThird = (vacationBase + avgOvertime + avgNight) / 3
      - abonoValue = dailyRate * abonoDays * 4/3 (exempt from INSS/IRRF per OJ 386)
      - grossTaxable = vacationBase + bonusThird + avgOvertime + avgNight
      - INSS via calculateINSS(grossTaxable, params.inssBrackets, params.inssCeiling)
      - IRRF via calculateIRRF({grossSalary: grossTaxable, inssContribution: inss, dependentCount, ...params})
      - fgts = grossTaxable * 0.08
      - net = grossTaxable - inss - irrf + abonoValue
    - `initVacationPeriod(employeeId, admissionDate, ctx)` — creates first ACCRUING period
    - `advancePeriod(periodId, ctx)` — transitions ACCRUING→AVAILABLE, creates next period
    - `scheduleVacation(input: ScheduleVacationInput, ctx)` — validates fractionation rules (min 5 days, max 3 fractions, one >= 14 days), calculates amounts, creates VacationSchedule, updates daysTaken on period
    - `cancelVacation(scheduleId, ctx)` — sets CANCELLED, returns days to period.daysTaken
    - `markAsPaid(scheduleId, ctx)` — transitions SCHEDULED→PAID, sets processedAt
    - `listAcquisitivePeriods(employeeId, orgId, ctx)` — includes balance (daysEarned - daysTaken), doublingDeadline (endDate + 12 months), isNearDoubling flag
    - `listSchedules(orgId, filters, ctx)` — filter by employeeId, status, date range
    - `getScheduleById(scheduleId, orgId, ctx)` — single schedule with employee name
    - `getVacationReceipt(scheduleId, orgId, ctx)` — generates PDF using pdfkit (vacation receipt)
    - `getExpiringPeriods(orgId, daysAhead)` — periods where doubling deadline - daysAhead <= today
    - Helper: `calcPaymentDueDate(startDate: Date): Date` — startDate minus 2 business days (skip weekends + holidays via date-holidays)
