**1. Tax guide alert cron (`tax-guide-alerts.cron.ts`):**

Follow the exact pattern from `contract-expiry-alerts.cron.ts`:

```typescript
import cron from 'node-cron';
import { prisma } from '../../lib/prisma';
import { redis } from '../../lib/redis';
import type { NotificationType } from '../../modules/notifications/notifications.types';

export function startTaxGuideAlertsCron(): void {
  // Run daily at 07:00
  cron.schedule('0 7 * * *', async () => {
    const today = new Date().toISOString().slice(0, 10);
    const lockKey = `cron:tax-guide-alerts:${today}`;
    const locked = await redis.set(lockKey, '1', 'EX', 3600, 'NX');
    if (!locked) return;

    try {
      const now = new Date();
      const in10days = new Date(now);
      in10days.setDate(in10days.getDate() + 10);

      // Find guides due within 10 days that are not PAID
      const guides = await prisma.taxGuide.findMany({
        where: {
          status: { notIn: ['PAID'] },
          dueDate: { gte: now, lte: in10days },
        },
        include: { organization: { select: { name: true } } },
      });

      const notificationType: NotificationType = 'TAX_GUIDE_DUE';

      for (const guide of guides) {
        const daysUntilDue = Math.ceil(
          (guide.dueDate.getTime() - now.getTime()) / (1000 * 60 * 60 * 24),
        );
        const alertLevel = daysUntilDue <= 5 ? 'DANGER' : 'WARNING';
        const message =
          daysUntilDue <= 5
            ? `Guia ${guide.guideType} vence em ${daysUntilDue} dia(s) — R$ ${guide.totalAmount}`
            : `Guia ${guide.guideType} vence em ${daysUntilDue} dias — R$ ${guide.totalAmount}`;

        // Create notification (same pattern as contract-expiry)
        await prisma.notification.create({
          data: {
            organizationId: guide.organizationId,
            type: notificationType,
            title: `Vencimento ${guide.guideType}`,
            message,
            severity: alertLevel === 'DANGER' ? 'HIGH' : 'MEDIUM',
            metadata: JSON.stringify({
              guideId: guide.id,
              guideType: guide.guideType,
              daysUntilDue,
            }),
          },
        });
      }
    } catch (err) {
      console.error('[tax-guide-alerts-cron] Error:', err);
    }
  });
}
```

**IMPORTANT:** Use `const notificationType: NotificationType = 'TAX_GUIDE_DUE'` — NEVER `as any`. The type is properly registered in step 0.
